文件选择器对话框

  • github.com/sqweek/dialog: 一个简易好用的跨平台的系统 对话框 API调用，可用于 选择文件（文件夹），消息提示，操作确认等。
package main

import (
    "fmt"
    sqdialog "github.com/sqweek/dialog"
)

func main() {
    filename, err := sqdialog.File().Filter("Excel表格(*.xlsx)", "xlsx").Load()
    if err != nil {
        panic(err)
    }
    fmt.Println(filename)
